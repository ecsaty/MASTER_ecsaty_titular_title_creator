namespace = ecsaty_create_titular

ecsaty_create_titular.0000 = {
	type = character_event
	title = ecsaty_create_titular.0000.t
	desc = ecsaty_create_titular.0000.desc
	theme = realm
	left_portrait = {
		character = root
		animation = war_defender
	}
	right_portrait = {
		character = scope:recipient
		animation = love
	}

	option = {
		name = ecsaty_create_titular.0000.me
		trigger_event = ecsaty_create_titular.0001
	}
	option = {
		trigger = {
			scope:recipient = {
				primary_title.tier >= tier_county
			}
		}
		name = ecsaty_create_titular.0000.recipient
		trigger_event = ecsaty_create_titular.0010
	}
	option = {
		name = ecsaty_create_titular.back
	}
}

ecsaty_create_titular.0002 = {
	hidden = yes
		
	immediate = {
		create_dynamic_title = {
			tier = duchy
			name = GET_TARGET_TITLE_NAME
		}

		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		scope:new_title = {
			set_color_from_title = scope:target
			set_coa = scope:target
			change_title_holder = {
				holder = root
				change = scope:change
			}
		}
		resolve_title_and_vassal_change = scope:change
	}
}
